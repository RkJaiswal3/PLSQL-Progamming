//System exception

DECLARE
x INT;
y INT;
result INT;
BEGIN 
x := 10;
y := 0;

result := x/y;

EXCEPTION
  WHEN ZERO_DIVIDE THEN
	DBMS_OUTPUT.PUT_LINE('Error: Division by zero is not allowed.'||SQLERRM);
  WHEN OTHERS THEN 
	DBMS_OUTPUT.PUT_LINE('Error: Which is not configure.');
END;

/
