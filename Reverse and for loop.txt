//using IN

create or replace procedure palidron(a in varchar2)
is
b varchar2(50);
begin


for i in Reverse 1..length(a) 
loop
 b := b || substr(a, i, 1);

end loop;

if lower(a) = lower(b) then
	dbms_output.put_line(a || ' is a palidrone');
else
dbms_output.put_line(a || 'is not a palidrone');

end if;

end;
/

begin
palidron('level');
end;

/
