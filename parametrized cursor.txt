DECLARE 
CURSOR abc(a number) IS
SELECT * FROM emp where sal > a;

my_row emp%ROWTYPE;

BEGIN
OPEN abc(5000);
LOOP
FETCH abc INTO my_row;
exit when abc%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(my_row.name);

END LOOP;

CLOSE abc;
END;
/
