// Record

// Table based record

declare 

my_row emp%rowtype;
begin 

select * into my_row from emp where id = 1;
dbms_output.put_line('Record : ' || my_row.name);

end;

/


// Cursor based record

declare
cursor get_name is select * from emp;

name_person get_name%ROWTYPE;

begin
open get_name;
loop
fetch get_name into name_person;
exit when get_name%notfound;
dbms_output.put_line('Record : ' || name_person.name);

end loop;

close get_name;

end;

/



// user defined record

declare

type name_person is record
(
	name_p varchar2(200),
	class_p number	

);
name_record name_person;

begin

select name, class into name_record from emp where id = 5;

dbms_output.put_line('Name : ' || name_record.name_p);
dbms_output.put_line('Class : ' || name_record.class_p);

end;
