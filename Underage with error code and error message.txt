declare
age int := 15;
underAge EXCEPTION;
overAge EXCEPTION;
begin
if age < 18 then
	raise_application_error(-20123, 'Under age'); 
end if;
if age > 60 then
	raise_application_error(-12223, 'Over age');
end if;

dbms_output.put_line('Your age is : ' || age);
	exception
	when others then
		dbms_output.put_line(SQLERRM);
		
end;
/
